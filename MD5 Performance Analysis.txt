# Function to print nicely formatted table
def print_table(results, headers):
    col_widths = [len(h) for h in headers]
    for row in results:
        for i, item in enumerate(row):
            col_widths[i] = max(col_widths[i], len(str(item)))
    
    header_row = "| " + " | ".join(h.ljust(col_widths[i]) for i, h in enumerate(headers)) + " |"
    separator_row = "|-" + "-|-".join("-" * col_widths[i] for i in range(len(headers))) + "-|"
    print(header_row)
    print(separator_row)
    
    for row in results:
        row_str = "| " + " | ".join(str(row[i]).rjust(col_widths[i]) if i != 1 else str(row[i]).ljust(col_widths[i]) for i in range(len(headers))) + " |"
        print(row_str)
