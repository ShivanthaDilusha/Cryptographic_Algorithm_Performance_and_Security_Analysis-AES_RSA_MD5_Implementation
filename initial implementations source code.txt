#!/usr/bin/env python3
"""
Week 9: Implementation and Initial Testing

This script demonstrates three cryptographic techniques:

1. AES (Symmetric Encryption) – AES-GCM mode
2. RSA (Asymmetric Encryption) – RSA-OAEP
3. MD5 (Hash Function) – hashing "Hello World"

Language   : Python 3.x
Dependency : cryptography
Install    : pip install cryptography
"""

import secrets
import time
import hashlib
import base64
from cryptography.hazmat.primitives.ciphers.aead import AESGCM
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.asymmetric import rsa, padding


# ---------------------------------------------------
# AES DEMO (Symmetric Key Encryption)
# ---------------------------------------------------
def aes_demo():
    print("\n--- AES-GCM DEMONSTRATION ---")

    # Generate a random 256-bit AES key
    key = AESGCM.generate_key(bit_length=256)
    aesgcm = AESGCM(key)

    # Nonce (random number required for AES-GCM)
    nonce = secrets.token_bytes(12)

    # Message to encrypt
    message = b"Hello World"

    # Encrypt
    start = time.perf_counter()
    ciphertext = aesgcm.encrypt(nonce, message, None)
    enc_time = time.perf_counter() - start

    # Decrypt
    start = time.perf_counter()
    decrypted = aesgcm.decrypt(nonce, ciphertext, None)
    dec_time = time.perf_counter() - start

    # Results
    print(f"Original message : {message.decode()}")
    print(f"Ciphertext (b64) : {base64.b64encode(ciphertext).decode()}")
    print(f"Decrypted text   : {decrypted.decode()}")
    print(f"Encryption time  : {enc_time:.6f} s")
    print(f"Decryption time  : {dec_time:.6f} s")


# ---------------------------------------------------
# RSA DEMO (Asymmetric Key Encryption)
# ---------------------------------------------------
def rsa_demo():
    print("\n--- RSA-OAEP DEMONSTRATION ---")

    # Generate RSA key pair (public and private)
    private_key = rsa.generate_private_key(
        public_exponent=65537,
        key_size=2048
    )
    public_key = private_key.public_key()

    message = b"Hello World"

    # Encrypt with public key
    start = time.perf_counter()
    ciphertext = public_key.encrypt(
        message,
        padding.OAEP(
            mgf=padding.MGF1(hashes.SHA256()),
            algorithm=hashes.SHA256(),
            label=None
        )
    )
    enc_time = time.perf_counter() - start

    # Decrypt with private key
    start = time.perf_counter()
    decrypted = private_key.decrypt(
        ciphertext,
        padding.OAEP(
            mgf=padding.MGF1(hashes.SHA256()),
            algorithm=hashes.SHA256(),
            label=None
        )
    )
    dec_time = time.perf_counter() - start

    # Results
    print(f"Original message : {message.decode()}")
    print(f"Ciphertext (b64) : {base64.b64encode(ciphertext).decode()}")
    print(f"Decrypted text   : {decrypted.decode()}")
    print(f"Encryption time  : {enc_time:.6f} s")
    print(f"Decryption time  : {dec_time:.6f} s")


# ---------------------------------------------------
# MD5 DEMO (Hash Function)
# ---------------------------------------------------
def md5_demo():
    print("\n--- MD5 DEMONSTRATION ---")

    message = b"Hello World"

    # Hashing
    start = time.perf_counter()
    digest = hashlib.md5(message).hexdigest()
    hash_time = time.perf_counter() - start

    # Results
    print(f"Message     : {message.decode()}")
    print(f"MD5 Digest  : {digest}")
    print(f"Hashing time: {hash_time:.6f} s")


# ---------------------------------------------------
# MAIN FUNCTION
# ---------------------------------------------------
def main():
    print("=== Week 9: Cryptography Implementation & Testing ===")
    aes_demo()
    rsa_demo()
    md5_demo()
    print("\nAll demonstrations completed successfully.")


if __name__ == "__main__":
    main()
